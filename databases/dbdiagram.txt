// =============================================
// Tables & Columns (DBML)
// =============================================

Table users {
  id bigint [pk, increment] // BIGSERIAL
  username varchar(50) [unique, not null]
  full_name varchar(200) [not null]
  created_at timestamptz [not null, default: `now()`]
  updated_at timestamptz [not null, default: `now()`]
  deleted_at timestamptz [default: null, note: 'soft delete (users are never physically removed)']
}

Table subscriptions {
  follower_id bigint [not null, ref: > users.id]
  followee_id bigint [not null, ref: > users.id]
  created_at timestamptz [not null, default: `now()`]

  indexes {
    (follower_id, followee_id) [unique, name: 'uq_subscription']
    followee_id [name: 'idx_subscriptions_followee']
  }
}

Table posts {
  id bigint [pk, increment]
  author_id bigint [not null, ref: > users.id]
  description varchar(500) [not null, note: 'CHECK char_length(description) <= 500']
  latitude decimal(9,6)
  longitude decimal(9,6)
  like_count int [not null, default: 0, note: 'denormalized counter']
  comment_count int [not null, default: 0, note: 'denormalized counter']
  created_at timestamptz [not null, default: `now()`]
  updated_at timestamptz [not null, default: `now()`]
  deleted_at timestamptz [default: null, note: 'soft delete']

  indexes {
    (author_id, created_at) [name: 'idx_posts_author_created', note: 'DESC on created_at (user profile feed)']
    (created_at) [name: 'idx_posts_created', note: 'DESC (global feed)']
    (deleted_at) [name: 'idx_posts_deleted', note: 'WHERE deleted_at IS NULL']
  }
}

Table likes {
  user_id bigint [not null, ref: > users.id]
  post_id bigint [not null, ref: > posts.id]
  created_at timestamptz [not null, default: `now()`]

  indexes {
    (post_id, user_id) [unique, name: 'uq_like']
    (user_id, created_at) [name: 'idx_likes_user_created', note: 'DESC on created_at']
  }
}

Table comments {
  id bigint [pk, increment]
  post_id bigint [not null, ref: > posts.id]
  author_id bigint [not null, ref: > users.id]
  text varchar(200) [not null, note: 'CHECK char_length(text) <= 200']
  created_at timestamptz [not null, default: `now()`]
  updated_at timestamptz [not null, default: `now()`]
  deleted_at timestamptz [default: null, note: 'soft delete']

  indexes {
    (post_id, created_at) [name: 'idx_comments_post_created', note: 'DESC on created_at (paginate comments)']
    (author_id, created_at) [name: 'idx_comments_author_created', note: 'DESC on created_at (user history)']
  }
}

Table photos {
  id bigint [pk, increment]
  post_id bigint [not null, ref: > posts.id]
  original_name varchar(500) [not null]
  s3_path varchar(1000) [not null, note: 'S3 object key']
  size_bytes bigint [not null]
  width int
  height int
  created_at timestamptz [not null, default: `now()`]
  updated_at timestamptz [not null, default: `now()`]
  deleted_at timestamptz [default: null]

  indexes {
    post_id [name: 'idx_photos_post']
  }
}
